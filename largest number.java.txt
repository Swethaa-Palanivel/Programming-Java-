import java.util.Scanner;

public class SumOfDigits {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int num = sc.nextInt();
        int max = 0;
        while (num > 0) {
            max = num% 10 > max ? num % 10 :max;
           num = num/10;
       
        }
        System.out.println(max);
    }
}
